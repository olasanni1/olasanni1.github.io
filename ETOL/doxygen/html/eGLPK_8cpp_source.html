192  int ind_i, int ind_j, int tIdx, std::vector<std::string> pnames,

196  std::vector<std::pair<std::vector<glpk_t>, glpk_bnd_t>> abs_obsv;
197  for (auto side : seg) {
198  std::string r(eGLPK::getParamName(
199  eGLPK::getParamName(name, ind_i, ind_j, ind_k), tIdx));
200  bool found(false);
201  int i = 0;
202  for (const auto& pname : pnames) {
203  if (re2::RE2::FullMatch(pname, r)) {
204  abs_obs.emplace_back(pname, tIdx, -1, sign*BigM);
205  found = true;
206  break;
207  }
208  i++;
209  }
210  if (!found)
211  break;
212  double m = seg.at(ind_k).second.slope;
213  abs_obs.emplace_back("x", tIdx, 0, -m); // x
214  abs_obs.emplace_back("x", tIdx, 1, 1); // y
215 
216  double xa = std::get<0>(seg.at(ind_k).first);
217  double ya = std::get<1>(seg.at(ind_k).first);
218  double rhs = ya - m * xa;
219  if (sign < 0)
220  abs_obsv.emplace_back(abs_obs, glpk_bnd_t(GLPK_UNBOUND, rhs));
221  else
222  abs_obsv.emplace_back(abs_obs, glpk_bnd_t(rhs, GLPK_UNBOUND));
223  abs_obs.clear();
224  ind_k++;
225  }
226  return abs_obsv;

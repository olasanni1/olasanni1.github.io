186 ETOL::f_t saaConstraint(ETOL::TrajectoryOptimizer* t) {
187  tracks = ETOL::eGLPK::createTracks(t, NSIDES);
188  ETOL::f_t saa = [](F_ARGS){
189  auto delta = std::any_cast<double>(dt);
190  int tIdx = std::any_cast<int>(k);
191  double t = delta * tIdx;
192  std::vector<ETOL::glpk_t> abs_saa;
193  std::vector<std::pair<std::vector<ETOL::glpk_t>,
194  ETOL::glpk_bnd_t>> abs_saav, abs_saav_sum;
195  int ii = 0;
196  for (auto track : tracks) {
197  double radius = track.radius;
198  std::vector<double> tvec;
199  for_each(track.trajectory.begin(), track.trajectory.end(), [&tvec]
200  (const ETOL::traj_elem_t& data){
201  double value = data.first;
202  tvec.push_back(value);
203  });
204  std::vector<double>::iterator tlo_it;
205  tlo_it = std::lower_bound(tvec.begin(), tvec.end(), t);
206  if (tlo_it != tvec.begin() && *tlo_it > t)
207  tlo_it = std::prev(tlo_it);
208  size_t jj = distance(tvec.begin(), tlo_it);
209 
210  ETOL::traj_t::iterator curr, next;
211  curr = track.trajectory.begin();
212  std::advance(curr, jj);
213  next = curr;
214  if (tlo_it != prev(tvec.end()))
215  std::advance(next, 1);
216 
217  double delt = next->first - curr->first;
218  double rhs;
219  for (int kk = 0; kk < NSIDES; kk++) {
220  std::string r(ETOL::eGLPK::getParamName(
221  ETOL::eGLPK::getParamName("d", ii, jj, kk), tIdx));
222  int i = 0;
223  for (const auto& pname : pnames) {
224  if (re2::RE2::FullMatch(pname, r)) {
225  abs_saa.emplace_back(pname, tIdx, -1, BIGM);
226  break;
227  }
228  i++;
229  }
230 
231  double c0 = cos(2. * M_PI * static_cast<double>(kk) /
232  static_cast<double>(NSIDES));
233  double c1 = sin(2. * M_PI * static_cast<double>(kk) /
234  static_cast<double>(NSIDES));
235 
236  double mx(0), my(0);
237  double valx0 = curr->second.at(0);
238  double valy0 = curr->second.at(1);
239  double valx1 = next->second.at(0);
240  double valy1 = next->second.at(1);
241 
242  if (delt != 0) {
243  mx = (valx1 - valx0)/ delt;
244  my = (valy1 - valy0)/ delt;
245  }
246  double cx = valx0 + mx*(t - *tlo_it);
247  double cy = valy0 + my*(t - *tlo_it);
248  rhs = c0*cx + c1*cy + radius;
249 
250  abs_saa.emplace_back("x", tIdx, 0, c0); // x
251  abs_saa.emplace_back("x", tIdx, 1, c1); // y
252 
253  abs_saav.emplace_back(abs_saa,
254  ETOL::glpk_bnd_t(rhs, GLPK_UNBOUND));
255  abs_saa.clear();
256  }
257  abs_saav_sum = ETOL::eGLPK::addObstacleSum("^d", ii, jj, tIdx,
258  pnames, NSIDES);
259  abs_saav.insert(abs_saav.end(), abs_saav_sum.begin(),
260  abs_saav_sum.end());
261  ii++;
262  }
263  return abs_saav;

37  bool ValidityChecker::doIntersect(Point p1, Point q1, Point p2,
38  Point q2) const {
39  // Find the four orientations needed for general and
40  // special cases
41  int o1 = orientation(p1, q1, p2);
42  int o2 = orientation(p1, q1, q2);
43  int o3 = orientation(p2, q2, p1);
44  int o4 = orientation(p2, q2, q1);
45 
46  // General case
47  if (o1 != o2 && o3 != o4)
48  return true;
49 
50  // Special Cases
51  // p1, q1 and p2 are colinear and p2 lies on segment p1q1
52  if (o1 == 0 && onSegment(p1, p2, q1)) return true;
53 
54  // p1, q1 and p2 are colinear and q2 lies on segment p1q1
55  if (o2 == 0 && onSegment(p1, q2, q1)) return true;
56 
57  // p2, q2 and p1 are colinear and p1 lies on segment p2q2
58  if (o3 == 0 && onSegment(p2, p1, q2)) return true;
59 
60  // p2, q2 and q1 are colinear and q1 lies on segment p2q2
61  if (o4 == 0 && onSegment(p2, q1, q2)) return true;
62 
63  return false; // Doesn't fall in any of the above cases
64  }

66  bool ValidityChecker::isInside(Point polygon[], int n, Point p) const {
67  // There must be at least 3 vertices in polygon[]
68  if (n < 3) return false;
69 
70  // Create a point for line segment from p to infinite
71  Point extreme = {INF, p.y};
72 
73  // Count intersections of the above line with sides of polygon
74  int count = 0, i = 0;
75  do {
76  int next = (i+1)%n;
77 
78  // Check if the line segment from 'p' to 'extreme' intersects
79  // with the line segment from 'polygon[i]' to 'polygon[next]'
80  if (doIntersect(polygon[i], polygon[next], p, extreme)) {
81  // If the point 'p' is colinear with line segment 'i-next',
82  // then check if it lies on segment. If it lies, return true,
83  // otherwise false
84  if (orientation(polygon[i], p, polygon[next]) == 0)
85  return onSegment(polygon[i], p, polygon[next]);
86  count++;
87  }
88  i = next;
89  } while (i != 0);
90 
91  // Return true if count is odd, false otherwise
92  return count&1; // Same as (count%2 == 1)
93  }

96  bool res = true;
